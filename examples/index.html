Toastbox Examples

Interactive demos of Toastbox features and functionality

Q: Wait, what's Toasbox? A: Well, it's a lightweight, modern alternative to native JavaScript alert, confirm, and prompt dialogs using Bootstrap 5

Alert Dialogs

Simple notification dialogs with an OK button. Perfect for showing information to users.

Basic Alert

toastbox.alert("Hello World! This is a basic alert dialog.");

Alert with Title

toastbox.alert({
  title: "Success!",
  message: "Cool. This alert has both a title and message."
});

Alert with Callback

toastbox.alert("This alert has a callback function that will execute when dismissed.", function() {
  // Callback executed when dismissed
});

Custom Button Text

toastbox.alert({
  message: "This alert has a custom button label instead of the default 'OK'.",
  buttons: { ok: { label: "Cool", className: "btn-success" } }
});

Confirm Dialogs

Decision dialogs with Cancel and OK buttons.

Basic Confirm

toastbox.confirm("Are you sure about this?", function(result) {
  // Handle result
});

Confirm with Title

toastbox.confirm({
  title: "Delete Item",
  message: "This action cannot be undone. Are you sure you want to delete this amazing item?",
  callback: function(result) { /* Handle result */ }
});

Custom Button Labels

toastbox.confirm({
  title: "Save Changes",
  message: "Uh oh. You have unsaved changes.",
  buttons: {
    cancel: { label: "Discard Changes", className: "btn-outline-secondary" },
    confirm: { label: "Save & Continue", className: "btn-primary" }
  },
  callback: function(result) { /* Handle result */ }
});

Dangerous Action

toastbox.confirm({
  title: "Super Dangerous Action",
  message: "This will create the Torment Nexus. This action can't be undone.",
  className: "text-danger",
  buttons: {
    cancel: { label: "Cancel", className: "btn-outline-secondary" },
    confirm: { label: "Activate Torment", className: "btn-danger" }
  },
  callback: function(result) { /* Handle result */ }
});

Prompt Dialogs

Input dialogs for collecting data. Supports various input types and validation.

Text Input

toastbox.prompt({
  title: "Enter Your Name",
  message: "Please enter your full dang name:",
  placeholder: "e.g. Homestar Runner",
  callback: function(result) { /* Handle result */ }
});

Email Input

toastbox.prompt({
  title: "Email Address",
  message: "Please enter your email address:",
  inputType: "email",
  placeholder: "DJmankiewicz@homestarrunnerr.com",
  callback: function(result) { /* Handle result */ }
});

Number Input

toastbox.prompt({
  title: "Enter Your Age",
  message: "Please enter your age in years:",
  inputType: "number",
  min: 1,
  max: 120,
  placeholder: "25",
  callback: function(result) { /* Handle result */ }
});

Textarea Input

toastbox.prompt({
  title: "Comments",
  message: "Please share mild feedback:",
  inputType: "textarea",
  placeholder: "Enter your highly tempered comments",
  rows: 4,
  callback: function(result) { /* Handle result */ }
});

Select Dropdown

toastbox.prompt({
  title: "Choose Your Plan",
  message: "Select a subscription plan:",
  inputType: "select",
  inputOptions: [
    { text: "Option 1 - Basic Plan", value: "basic" },
    { text: "Option 2 - Expensive Plan", value: "premium" },
    { text: "Option 3 - Outrageously Expensive Plan", value: "enterprise" }
  ],
  callback: function(result) { /* Handle result */ }
});

Multiple Checkboxes

toastbox.prompt({
  title: "Notification Preferences",
  message: "Select which notifications you'd like to receive:",
  inputType: "checkbox",
  inputOptions: [
    { text: "Email notifications", value: "email" },
    { text: "SMS notifications", value: "sms" },

  callback: function(result) { /* Handle result */ }
});

Custom Dialogs

Flexible dialogs with custom buttons, styling, and behavior.

Custom Buttons

toastbox.dialog({
  title: "Choose Your Fate",
  message: "What next?",
  buttons: {
    save: { label: "Save", className: "btn-success", callback: function() {} },
    edit: { label: "Edit", className: "btn-warning", callback: function() {} },
    delete: { label: "Delete", className: "btn-danger", callback: function() {} },
    cancel: { label: "Cancel", className: "btn-outline-secondary", callback: function() {} }
  }
});

Large Dialog

toastbox.dialog({
  title: "Large Dialog",
  message: "<h5>This is a large dialog</h5><p>Large dialogs are cool for displaying more content, forms, or detailed information.</p>",
  size: "large",
  buttons: { ok: { label: "Close", className: "btn-primary" } }
});

Small Dialog

toastbox.dialog({
  title: "Small Dialog",
  message: "This is a compact dialog for quick notifications or simple confirmations.",
  size: "small",
  buttons: { ok: { label: "OK", className: "btn-primary" } }
});

Centered Dialog

toastbox.dialog({
  title: "Centered Dialog",
  message: "This dialog is vertically centered on the screen. Neat.",
  centerVertical: true,
  buttons: { ok: { label: "Nice!", className: "btn-primary" } }
});

Configuration & Features

Advanced features and configuration options.

Disable Animation

toastbox.alert({
  title: "No Animation",
  message: "This dialog appears instantly without the fade-in animation.",
  animate: false
});

Custom CSS Classes

toastbox.alert({
  title: "Custom Styling",
  message: "This dialog has custom CSS classes applied for unique styling.",
  className: "border-success shadow-lg",
  buttons: { ok: { label: "Stylish!", className: "btn-success" } }
});

Without Close Button

toastbox.alert({
  title: "No Close Button??",
  message: "This dialog doesn't have the X close button in the header. Users must use the OK button to dismiss it. Wow.",
  closeButton: false
});

Backdrop Options

toastbox.alert({
  title: "Backdrop Demo",
  message: "Try clicking on the background (backdrop) behind this dialog. It should dismiss the dialog.",
  backdrop: true,
  onHide: function() { /* Handle hide event */ }
});
